88  auto trans_z = transformStamped.transform.translation.z;
89 
90  tf::Quaternion q(transformStamped.transform.rotation.x, transformStamped.transform.rotation.y, transformStamped.transform.rotation.z, transformStamped.transform.rotation.w);
91  double roll, pitch, yaw;
92  tf::Matrix3x3 m(q);
93  m.getRPY(roll, pitch, yaw);
94 
95  ROS_INFO_STREAM("Position in map frame: ["
96  << trans_x << ","
97  << trans_y << ","
98  << trans_z << "]"
99  );
100  ROS_INFO_STREAM("Fiducial ID: " << m_aruco_fiducial_id);
101 
102  //offset the coordinates for the follower to reach
103  double offset = 0.4;
104  double theta = -3.1417/2 + yaw;
105 
106  //Store the transformed target location
107  m_follower_locations[m_aruco_fiducial_id][0]= trans_x + offset*cos(theta);//fiducial id acts as array index to store target locations in order
108  m_follower_locations[m_aruco_fiducial_id][1]= trans_y + offset*sin(theta);
109 
110  }
